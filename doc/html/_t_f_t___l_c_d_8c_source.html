308 void LCD_flood (uint16_t color, uint32_t length)
309 {
310  uint16_t blocks;
311  uint8_t i, color_high = color >> 8, color_low = color;
312 
313  /* Write first pixel normally, decrement counter by 1 */
314  LCD_RS_data();
315  TFT_LCD_transmit_8bits(color_high);
316  TFT_LCD_transmit_8bits(color_low);
317  length--;
318 
319  blocks = (uint16_t)(length / 64); /* 64 pixels/block */
320 
321  /* High and low bytes are identical. Leave prior data on the port(s) and just toggle the write strobe. */
322  if (color_high == color_low)
323  {
324  while(blocks--)
325  {
326  i = 16; /* 64 pixels/block / 4 pixels/pass */
327  do
328  {
329  TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low); TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low);
330  TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low); TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low);
331  }
332  while(--i);
333  }
334  /* Fill any remaining pixels (1 to 64) */
335  for (i = (uint8_t)length & 63; i--;)
336  {
337  TFT_LCD_transmit_8bits(color_high);
338  TFT_LCD_transmit_8bits(color_low);
339  }
340  }
341  else
342  {
343  while(blocks--)
344  {
345  i = 16; /* 64 pixels/block / 4 pixels/pass */
346  do
347  {
348  TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low); TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low);
349  TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low); TFT_LCD_transmit_8bits(color_high); TFT_LCD_transmit_8bits(color_low);
350  }
351  while(--i);
352  }
353  /* Fill any remaining pixels (1 to 64) */
354  for (i = (uint8_t)length & 63; i--;)
355  {
356  TFT_LCD_transmit_8bits(color_high);
357  TFT_LCD_transmit_8bits(color_low);
358  }
359  }
360 }

387 void LCD_draw_line (int16_t x0, int16_t y0, int16_t x1, int16_t y1, uint16_t color)
388 {
389  int16_t steep = abs(y1 - y0) > abs(x1 - x0);
390 
391  if (steep)
392  {
393  swap(x0, y0);
394  swap(x1, y1);
395  }
396 
397  if (x0 > x1)
398  {
399  swap(x0, x1);
400  swap(y0, y1);
401  }
402 
403  int16_t dx, dy;
404  dx = x1 - x0;
405  dy = abs(y1 - y0);
406 
407  int16_t err = dx / 2;
408  int16_t y_step;
409 
410  if (y0 < y1)
411  {
412  y_step = 1;
413  }
414  else
415  {
416  y_step = -1;
417  }
418 
419  for (; x0 <= x1; x0++)
420  {
421  if (steep)
422  {
423  LCD_draw_pixel(y0, x0, color);
424  }
425  else
426  {
427  LCD_draw_pixel(x0, y0, color);
428  }
429  err -= dy;
430  if (err < 0)
431  {
432  y0 += y_step;
433  err += dx;
434  }
435  }
436 }

446 void LCD_draw_fast_horizontal_line (int16_t x, int16_t y, int16_t length, uint16_t color)
447 {

450  /* Check if the desired position is valid. Off-screen clipping */

452 
453  if (x < 0) /* Clip left */
454  {
455  length += x;
456  x = 0;
457  }

918 void LCD_fill_triangle (int16_t x0, int16_t y0, int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint16_t color)
919 {
920  int16_t a, b, y, last;
921 
922  /* Sort coordinates by Y order (y2 >= y1 >= y0) */
923  if (y0 > y1)
924  {
925  swap(y0, y1);
926  swap(x0, x1);
927  }
928 
929  if (y1 > y2)
930  {
931  swap(y2, y1);
932  swap(x2, x1);
933  }
934 
935  if (y0 > y1)
936  {
937  swap(y0, y1);
938  swap(x0, x1);
939  }
940 
941  /* Handle awkward all-on-same-line case as its own thing */
942  if(y0 == y2)
943  {
944  a = b = x0;
945  if(x1 < a) a = x1;
946  else if(x1 > b) b = x1;
947  if(x2 < a) a = x2;
948  else if(x2 > b) b = x2;
949  LCD_draw_fast_horizontal_line(a, y0, b-a+1, color);
950  return;
951  }
952 
953  int16_t
954  dx01 = x1 - x0,
955  dy01 = y1 - y0,
956  dx02 = x2 - x0,
957  dy02 = y2 - y0,
958  dx12 = x2 - x1,
959  dy12 = y2 - y1,
960  sa = 0,
961  sb = 0;
962 
963  /* For upper part of triangle, find scan-line crossings for segments
964  * 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scan-line y1
965  * is included here (and second loop will be skipped, avoiding a /0
966  * error there), otherwise scan-line y1 is skipped here and handled
967  * in the second loop...which also avoids a /0 error here if y0=y1
968  * (flat-topped triangle).
969  */
970  if(y1 == y2) last = y1; /* Include y1 scan-line */
971  else last = y1 - 1; /* Skip it */
972 
973  for(y = y0; y <= last; y++)
974  {
975  a = x0 + sa / dy01;
976  b = x0 + sb / dy02;
977  sa += dx01;
978  sb += dx02;
979  /* long-hand:
980  * a = x0 + (x1 - x0) * (y - y0) / (y1 - y0);
981  * b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
982  */
983  if(a > b) swap(a, b);
984  LCD_draw_fast_horizontal_line(a, y, b - a + 1, color);
985  }
986 
987  /* For lower part of triangle, find scan-line crossings for segments
988  * 0-2 and 1-2. This loop is skipped if y1=y2.
989  */
990  sa = dx12 * (y - y1);
991  sb = dx02 * (y - y0);
992 
993  for(; y <= y2; y++) {
994  a = x1 + sa / dy12;
995  b = x0 + sb / dy02;
996  sa += dx12;
997  sb += dx02;
998  /* long-hand:
999  * a = x1 + (x2 - x1) * (y - y1) / (y2 - y1);
1000  * b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
1001  */
1002  if(a > b) swap(a, b);
1003  LCD_draw_fast_horizontal_line(a, y, b - a + 1, color);
1004  }
1005 }
