46 void FLEXCAN0_init(void) {
47 #define MSG_BUF_SIZE 18 /* Msg Buffer Size. (2 words hdr + 16 words data = 18 words) */
48  uint32_t i=0;
49 
50  PCC->PCCn[PCC_FlexCAN0_INDEX] |= PCC_PCCn_CGC_MASK; /* CGC=1: enable clock to FlexCAN0 */
51  CAN0->MCR |= CAN_MCR_MDIS_MASK; /* MDIS=1: Disable module before selecting clock */
52  CAN0->CTRL1 |= CAN_CTRL1_CLKSRC_MASK; /* CLKSRC=1: Clock Source = BUSCLK (40 MHz) */
53  CAN0->MCR &= ~CAN_MCR_MDIS_MASK; /* MDIS=0; Enable module config. (Sets FRZ, HALT)*/
54  while (!((CAN0->MCR & CAN_MCR_FRZACK_MASK) >> CAN_MCR_FRZACK_SHIFT)) {}
55 
56  /* Good practice: wait for FRZACK=1 on freeze mode entry/exit */
57 
58  CAN0->CBT = CAN_CBT_EPSEG2(15) /* Configure nominal phase: 500 KHz bit time, 40 MHz Sclock */
59  |CAN_CBT_EPSEG1(15) /* Prescaler = CANCLK / Sclock = 40 MHz / 40 MHz = 1 */
60  |CAN_CBT_EPROPSEG(47) /* EPRESDIV = Prescaler - 1 = 0 */
61  |CAN_CBT_ERJW(15); /* EPSEG2 = 15 */
62  /* EPSEG1 = 15 */
63  /* EPROPSEG = 47 */
64  /* ERJW = 15 */
65  /* BITRATEn =Fcanclk /( [(1 + (EPSEG1+1) + (EPSEG2+1) + (EPROPSEG + 1)] x (EPRESDIV+1)) */
66  /* = 40 MHz /( [(1 + ( 15 +1) + ( 15 +1) + ( 46 + 1)] x ( 0 +1)) */
67  /* = 40 MHz /( [1+16+16+47] x 1) = 40 MHz /(80x1) = 500 Kz */
68 
69  /* Configure data phase: 2 MHz bit time, 40 MHz Sclock */
70  CAN0->FDCBT = CAN_CBT_EPSEG2(3)/* Prescaler = CANCLK / Sclock = 40 MHz / 40 MHz = 1 */
71  |CAN_CBT_EPSEG1(7)/* FPRESDIV = Prescaler - 1 = 0 */
72  |CAN_CBT_EPROPSEG(7)/* FPSEG2 = 3 */
73  |CAN_CBT_ERJW(3);/* FPSEG1 = 7 */
74  /* FPROPSEG = 7 */
75  /* FRJW = 3 */
76  /* BITRATEf =Fcanclk /( [(1 + (FPSEG1+1) + (FPSEG2+1) + (FPROPSEG)] x (FPRESDIV+!)) */
77  /* = 40 MHz /( [(1 + ( 7 +1) + ( 3 +1) + ( 7 )] x ( 0 +1)) */
78  /* = 40 MHz /( [1+8+4+7] x 1) = 40 MHz /(20x1) = 40 MHz / 20 = 2 MHz */
79 
80 
81  CAN0->FDCTRL = CAN_FDCTRL_FDRATE_MASK /* Configure bit rate switch, data size, transcv'r delay */
82  |CAN_FDCTRL_MBDSR0(3) /* BRS=1: enable Bit Rate Swtich in frame's header */
83  |CAN_FDCTRL_TDCEN_MASK /* MBDSR1: Not applicable */
84  |CAN_FDCTRL_TDCOFF(5); /* MBDSR0=3: Region 0 has 64 bytes data in frame's payload */
85  /* TDCEN=1: enable Transceiver Delay Compensation */
86  /* TDCOFF=5: 5 CAN clocks (300us) offset used */
87 
88  for(i=0; i<128; i++ ) { /* CAN0: clear 128 words RAM in FlexCAN 0 */
89  CAN0->RAMn[i] = 0; /* Clear msg buf words. All buffers CODE=0 (inactive) */
90  }
91  for(i=0; i<16; i++ ) { /* In FRZ mode, init CAN0 16 msg buf filters */
92  CAN0->RXIMR[i] = 0xFFFFFFFF; /* Check all ID bits for incoming messages */
93  }
94  CAN0->RXMGMASK = 0x1FFFFFFF; /* Global acceptance mask: check all ID bits */
95 
96  /* Message Buffer 4 - receive setup: */
97  CAN0->RAMn[ 4*MSG_BUF_SIZE + 0] = 0xC4000000; /* Msg Buf 4, word 0: Enable for reception */
98  /* EDL=1: Extended Data Length for CAN FD */
99  /* BRS = 1: Bit Rate Switch enabled */
100  /* ESI = 0: Error state */
101  /* CODE=4: MB set to RX inactive */
102  /* IDE=0: Standard ID */
103  /* SRR, RTR, TIME STAMP = 0: not applicable */
104 #ifdef NODE_A /* Node A receives msg with std ID 0x511 */
105  CAN0->RAMn[ 4*MSG_BUF_SIZE + 1] = 0x14440000; /* Msg Buf 4, word 1: Standard ID = 0x511 */
106 #else /* Node B to receive msg with std ID 0x555 */
107  CAN0->RAMn[ 4*MSG_BUF_SIZE + 1] = 0x15540000; /* Msg Buf 4, word 1: Standard ID = 0x555 */
108 #endif
109  /* PRIO = 0: CANFD not used */
110  CAN0->CTRL2 |= CAN_CTRL2_ISOCANFDEN_MASK; /* Enable CRC fix for ISO CAN FD */
111  CAN0->MCR = 0x0000081F; /* Negate FlexCAN 1 halt state & enable CAN FD for 32 MBs */
112  while ((CAN0->MCR && CAN_MCR_FRZACK_MASK) >> CAN_MCR_FRZACK_SHIFT) {}
113  /* Good practice: wait for FRZACK to clear (not in freeze mode) */
114  while ((CAN0->MCR && CAN_MCR_NOTRDY_MASK) >> CAN_MCR_NOTRDY_SHIFT) {}
115  /* Good practice: wait for NOTRDY to clear (module ready) */
116 }

118 void FLEXCAN0_transmit_msg(void) {
119 
125 /* Assumption: Message buffer CODE is INACTIVE */
126  CAN0->IFLAG1 = 0x00000001; /* Clear CAN 0 MB 0 flag without clearing others*/
127  CAN0->RAMn[ 0*MSG_BUF_SIZE + 2] = 0xA5112233; /* MB0 word 2: data word 0 */
128  CAN0->RAMn[ 0*MSG_BUF_SIZE + 3] = 0x44556677; /* MB0 word 3: data word 1 */
129 #ifdef NODE_A
130  CAN0->RAMn[ 0*MSG_BUF_SIZE + 1] = 0x15540000; /* MB0 word 1: Tx msg with STD ID 0x555 */
131 #else
132  CAN0->RAMn[ 0*MSG_BUF_SIZE + 1] = 0x14440000; /* MB0 word 1: Tx msg with STD ID 0x511 */
133 #endif
134  CAN0->RAMn[ 0*MSG_BUF_SIZE + 0] = 0xCC4F0000 | 8 <<CAN_WMBn_CS_DLC_SHIFT; /* MB0 word 0: */
135  /* EDL=1 CAN FD format frame*/
136  /* BRS=1: Bit rate is switched inside msg */
137  /* ESI=0: ??? */
138  /* CODE=0xC: Activate msg buf to transmit */
139  /* IDE=0: Standard ID */
140  /* SRR=1 Tx frame (not req'd for std ID) */
141  /* RTR = 0: data, not remote tx request frame*/
142  /* DLC=15; 64 bytes */
143 }

145 void FLEXCAN0_receive_msg(void) {
149  uint8_t j;
150  uint32_t dummy;
151 
152  RxCODE = (CAN0->RAMn[ 4*MSG_BUF_SIZE + 0] & 0x07000000) >> 24; /* Read CODE field */
153  RxID = (CAN0->RAMn[ 4*MSG_BUF_SIZE + 1] & CAN_WMBn_ID_ID_MASK) >> CAN_WMBn_ID_ID_SHIFT ;
154  RxLENGTH = (CAN0->RAMn[ 4*MSG_BUF_SIZE + 0] & CAN_WMBn_CS_DLC_MASK) >> CAN_WMBn_CS_DLC_SHIFT;
155  for (j=0; j<2; j++) { /* Read two words of data (8 bytes) */
156  RxDATA[j] = CAN0->RAMn[ 4*MSG_BUF_SIZE + 2 + j];
157  }
158  RxTIMESTAMP = (CAN0->RAMn[ 0*MSG_BUF_SIZE + 0] & 0x000FFFF);
159  dummy = CAN0->TIMER; /* Read TIMER to unlock message buffers */
160  CAN0->IFLAG1 = 0x00000010; /* Clear CAN 0 MB 4 flag without clearing others*/
161 }
