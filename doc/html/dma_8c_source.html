166 void DMA_Config(uint32_t Destination[4]){
167  SIM->PLATCGC |= SIM_PLATCGC_CGCDMA_MASK; /* DMA Clock Gating Control Enable */
168 
169  PCC->PCCn[PCC_DMAMUX_INDEX] |= PCC_PCCn_CGC_MASK; /* Enable DMA Clock */
170 
171  DMAMUX->CHCFG[3] &=~ DMAMUX_CHCFG_ENBL_MASK; /* Disabling the DMA channel 3 */
172  DMAMUX->CHCFG[3] |= DMAMUX_CHCFG_SOURCE(42); /* ADC0 COCO is the source of the DMA channel 3 */
173  DMAMUX->CHCFG[3] |= DMAMUX_CHCFG_ENBL_MASK; /* Enable the DMA channel 3 */
174 
175  /* Configure TCD*/
176  //Source
177  DMA->TCD[3].SADDR = (uint32_t)&ADC0->R[4]; /* Data Source */
178  DMA->TCD[3].SOFF = DMA_TCD_SOFF_SOFF(2); /* Offset Source 1 Byte */
179  DMA->TCD[3].SLAST = DMA_TCD_SLAST_SLAST(-20); /* An adjust of -5 is needed because are 4 channels*/
180 
181  //Destination
182  DMA->TCD[3].DADDR = (uint32_t)&Destination[0]; /* Data Destination */
183  DMA->TCD[3].DOFF = DMA_TCD_DOFF_DOFF(2); /* Offset Source 1 Byte */
184  DMA->TCD[3].DLASTSGA = DMA_TCD_DLASTSGA_DLASTSGA(-20); /* An adjust of -5 is needed */
185 
186  DMA->TCD[3].ATTR = DMA_TCD_ATTR_SSIZE(1)| /* 16 bits transfer from Source */
187  DMA_TCD_ATTR_DSIZE(1); /* 16 bits receive to Destination */
188 
189  DMA->TCD[3].NBYTES.MLOFFNO = DMA_TCD_NBYTES_MLOFFYES_NBYTES(2); /* Minor Byte Transfer Count is 2-bytes */
190  DMA->TCD[3].CITER.ELINKYES = DMA_TCD_CITER_ELINKYES_CITER_LE(10); /* Current Major Iteration Count is 1 */
191  DMA->TCD[3].BITER.ELINKYES = DMA_TCD_BITER_ELINKYES_BITER(10);
192 
193  DMA->TCD[3].CSR = DMA_TCD_CSR_MAJORELINK_MASK| /* The channel-to-channel linking is enable */
194  DMA_TCD_CSR_MAJORLINKCH(1); /* The channel is not explicitly started*/
195 
196  DMA->ERQ |= DMA_ERQ_ERQ3_MASK; /* The DMA request signal for CH3 is enabled */
197 
198 }
