129 void LPSPI0_init_master (void)
130 {
131  /* Configure Clock for LPSPI0 */
132  PCC -> PCCn[PCC_LPSPI0_INDEX] &= ~PCC_PCCn_CGC_MASK; /* CGC = 0 Clock disabled to modify PCS */
133  PCC -> PCCn[PCC_LPSPI0_INDEX] &= ~PCC_PCCn_PCS_MASK; /* Clear PCS field temporarily */
134  PCC -> PCCn[PCC_LPSPI0_INDEX] |= PCC_PCCn_PCS(6); /* PCS = 6 SPLLDIV2_CLK is selected */
135  PCC -> PCCn[PCC_LPSPI0_INDEX] |= PCC_PCCn_CGC_MASK; /* CGC = 1 Clock enabled for LPSPI0 */
136 
137  /* Configure as master mode */
138  LPSPI0 -> CFGR1 |= LPSPI_CFGR1_MASTER_MASK; /* MASTER = 1 Master mode */
139 
140  LPSPI0 -> TCR = LPSPI_TCR_PRESCALE(2) /* Divide by 4, = 10 MHz */
141  | LPSPI_TCR_FRAMESZ(7); /* Configure TCR, FRAMESZ = 7 Frame size is 8bit */
142 
143  LPSPI0 -> CCR |= LPSPI_CCR_SCKDIV(8) /* SCKDIV = 8 SCK cycle is 8+2 = 10 cycles of 10MHz thus 1MHz */
144  | LPSPI_CCR_SCKPCS(4) /* SCKPCS = 4 SCK to PCS Delay is (4 + 1) = 5 baud clock cycles */
145  | LPSPI_CCR_PCSSCK(4) /* PCSSCK = 4 PCS to SCK Delay is (4 + 1) = 5 baud clock cycles */
146  | LPSPI_CCR_DBT(8); /* DBT = 8 Delay Between Transfers is (8 + 2) = 10 baud clock cycles */
147 
148  LPSPI0 -> CR |= LPSPI_CR_MEN_MASK; /* MEN = 1 Module is enabled */
149 
150  LPSPI0 -> FCR &= ~LPSPI_FCR_RXWATER_MASK; /* RXWATER = 0 The Receive Data Flag is set whenever the number of words in the receive FIFO is greater than 0 */
151  LPSPI0 -> FCR &= ~LPSPI_FCR_TXWATER_MASK; /* TXWATER = 0 The Transmit Data Flag is set whenever the number of words in the transmit FIFO is equal or less than 0 */
152 
153 }
