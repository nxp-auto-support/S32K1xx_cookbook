39 void ADC_init (void)
40 {
41  /* ADC0 Clocking */
42  PCC -> PCCn[PCC_ADC0_INDEX] &= ~PCC_PCCn_CGC_MASK; /* Disable clock to change PCS */
43  PCC -> PCCn[PCC_ADC0_INDEX] |= PCC_PCCn_PCS(1); /* PCS = 1 Select SOSCDIV2 */
44  PCC -> PCCn[PCC_ADC0_INDEX] |= PCC_PCCn_CGC_MASK; /* Enable bus clock in ADC */
45 
46  /* ADC0 Initialization */
47  ADC0 -> SC1[0] |= ADC_SC1_ADCH_MASK; /* ADCH = 1F Module is disabled for conversions */
48  /* AIEN = 0 Interrupts are disabled */
49 
50  ADC0 -> CFG1 |= ADC_CFG1_ADIV_MASK /* ADIV = 0 Prescaler = 1 */
51  | ADC_CFG1_MODE(1); /* MODE = 1 12-bit conversion */
52  /* ADICLK = 0 Input CLK = ALTCLK1 = SOSCDIV2 */
53 
54  ADC0 -> CFG2 = ADC_CFG2_SMPLTS(12); /* SMPLTS = 12(default): sample time is 13 ADC clks */
55 
56  ADC0 -> SC2 = 0x00000000; /* ADTRG = 0 SW trigger */
57  /* ACFE, ACFGT, ACREN = 0 Compare functionality disabled */
58  /* DMAEN = 0 DMA disabled */
59  /* REFSEL = 0 Voltage reference pins= VREFH, VREEFL */
60 
61  ADC0 -> SC3 = 0x00000000; /* CAL = 0 Do not start calibration sequence */
62  /* ADCO = 0 One conversion performed */
63  /* AVGE, AVGS = 0 HW average function disabled */
64 }

69 void ADC_HW_trigger_init (int8_t adc_channel)
70 {
71  /* ADC0 Clocking */
72  PCC -> PCCn[PCC_ADC0_INDEX] &= ~PCC_PCCn_CGC_MASK; /* Disable clock to change PCS */
73  PCC -> PCCn[PCC_ADC0_INDEX] |= PCC_PCCn_PCS(1); /* PCS = 1 Select SOSCDIV2 */
74  PCC -> PCCn[PCC_ADC0_INDEX] |= PCC_PCCn_CGC_MASK; /* Enable bus clock in ADC */
75 
76  ADC0 -> SC1[0] |= ADC_SC1_ADCH_MASK /* ADCH = 1F Module is disabled for conversions */
77  | ADC_SC1_AIEN_MASK; /* AIEN = 1 Interrupts are enabled */
78 
79  ADC0 -> CFG1 |= ADC_CFG1_ADIV_MASK /* ADIV = 0 Prescaler = 1 */
80  | ADC_CFG1_MODE(1); /* MODE = 1 12-bit conversion */
81  /* ADICLK = 0 Input CLK = ALTCLK1 = SOSCDIV2 */
82 
83  ADC0 -> CFG2 = 0x00000000C; /* SMPLTS = 12(default): sample time is 13 ADC clks */
84 
85  ADC0 -> SC2 = ADC_SC2_ADTRG_MASK; /* ADTRG = 1 HW trigger */
86  /* ACFE, ACFGT, ACREN = 0: Compare functionality disabled */
87  /* DMAEN = 0 DMA disabled */

140 void ADC_calibration_init (int16_t gain, int16_t offset)
141 {
142  /* ADC0 Clocking */
143  PCC -> PCCn[PCC_ADC0_INDEX] &= ~PCC_PCCn_CGC_MASK; /* Disable clock to change PCS */
144  PCC -> PCCn[PCC_ADC0_INDEX] |= PCC_PCCn_PCS(1); /* PCS = 1 Select SOSCDIV2 */
145  PCC -> PCCn[PCC_ADC0_INDEX] |= PCC_PCCn_CGC_MASK; /* Enable bus clock in ADC */
146 
147  /* ADC0 Calibration */
148  ADC0 -> CFG1 |= ADC_CFG1_ADIV(2) /* ADIV = 0 Prescaler = 1 */
149  | ADC_CFG1_MODE(1); /* MODE = 1 12-bit conversion */
150  /* ADICLK = 0 Input CLK = ALTCLK1 = SOSCDIV2 */
151 
152  ADC0 -> SC3 |= ADC_SC3_AVGE_MASK /* AVGE = 1 Enable hardware average */
153  | ADC_SC3_AVGS(3); /* AVGS = 11b 32 samples averaged */
154 
155  /* Initialization for the Internal ADC0 Calibration Registers */
156  ADC0 -> CLPS = 0x00u;
157  ADC0 -> CLP3 = 0x00u;
158  ADC0 -> CLP2 = 0x00u;
159  ADC0 -> CLP1 = 0x00u;
160  ADC0 -> CLP0 = 0x00u;
161  ADC0 -> CLPX = 0x00u;
162  ADC0 -> CLP9 = 0x00u;
163  ADC0 -> USR_OFS = offset;
164  ADC0 -> UG = gain;
165 
166  ADC0 -> SC3 |= ADC_SC3_CAL_MASK; /* CAL = 1 Start calibration sequence */
167 
168  while(ADC_conversion_complete() == 0); /* Wait for conversion complete flag */
169 
170  /* ADC0 Initialization */
171  ADC0 -> SC1[0] |= ADC_SC1_ADCH_MASK; /* ADCH = 1F Module is disabled for conversions */
172  /* AIEN = 0 Interrupts are disabled */
173 
174  ADC0 -> CFG1 |= ADC_CFG1_ADIV_MASK /* ADIV = 0 Prescaler = 1 */
175  | ADC_CFG1_MODE(1); /* MODE = 1 12-bit conversion */
176  /* ADICLK = 0 Input CLK = ALTCLK1 = SOSCDIV2 */
177

179

182  /* DMAEN = 0 DMA disabled */
