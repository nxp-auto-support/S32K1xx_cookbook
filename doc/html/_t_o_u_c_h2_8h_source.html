65 /*****************************************************************************
66 *
67 * Function: static inline void ChargeDistribution2 (void)
68 * Description: Charge distribution
69 *
70 *****************************************************************************/
71 static inline void ChargeDistribution2 (void)
72 {
73  // Drive electrode GPIO low
74  ELEC2_GPIO->PCOR = 1 << ELEC2_ELEC_GPIO_PIN;
75 
76  // Configure electrode pin as GPIO to precharge electrode
77  ELEC2_PORT->PCR[ELEC2_ELEC_GPIO_PIN] = PCR_GPIO2;
78 
79  // Drive Cext GPIO high
80  ELEC2_GPIO->PSOR = 1 << ELEC2_CEXT_GPIO_PIN;
81 
82  // Configure Cext pin as GPIO to precharge Cext
83  ELEC2_PORT->PCR[ELEC2_CEXT_GPIO_PIN] = PCR_GPIO2;
84 
85  // Configure Electrode and Cext pins as outputs at the same time
86  REG_WRITE32(&(ELEC2_GPIO->PDDR), ((REG_READ32(&(ELEC2_GPIO->PDDR))) | (ELEC2_PORT_MASK)));
87 }
