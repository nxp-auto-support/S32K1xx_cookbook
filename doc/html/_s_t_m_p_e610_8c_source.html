48 void STMPE610_write_register (uint8_t reg, uint8_t val)
49 {

64 uint8_t STMPE610_read_register (uint8_t reg)
65 {
66  uint8_t temp;
67 
68  /* Read a register from the STMPE610 controller */
69  STMPE610_CS_low(); /* CS is active */
70  LPSPI0_write_8bits(0x80 | reg);
71  LPSPI0_write_8bits(0x00);
72  temp = LPSPI0_write_8bits(STMPE_NOP);
73  STMPE610_CS_high(); /* CS is inactive */
74 
75  return temp;
76 }

84 uint8_t STMPE610_check_version (void)
85 {
86  /* If LSB byte does not match with STMPE610 version, return false */
87  if (STMPE_MSB_VERSIONSUPPORTED == STMPE610_read_register(0x00));
88  else
89  {
90  return 0;
91  }
92 
93  /* If MSB byte does not match with STMPE610 version, return false */
94  if (STMPE_LSB_VERSIONSUPPORTED == STMPE610_read_register(0x01));
95  else
96  {
97  return 0;
98  }
99 
100  /* Version was successfully authenticated */
101  return 1;
102 }

112 uint8_t STMPE610_init (void)
113 {
114  uint8_t i;
115 
116  LPSPI0_init_master();
117 
118  /* Verify STMPE610 controller */
119  if (!STMPE610_check_version())
120  {
121  return 0;
122  }
123 
124  /* Reset STMPE610 */
125  STMPE610_write_register(STMPE_SYS_CTRL1, STMPE_SYS_CTRL1_RESET);
126 
127  /* Flush all registers after software reset */
128  for(i = 0; i < 65; i++)
129  {
130  STMPE610_read_register(i);
131  }
132 
133  /* STMPE610 initialization */
134  STMPE610_write_register(STMPE_SYS_CTRL2, 0x0);
135  STMPE610_write_register(STMPE_TSC_CTRL, STMPE_TSC_CTRL_XYZ | STMPE_TSC_CTRL_EN);
136  STMPE610_write_register(STMPE_INT_EN, STMPE_INT_EN_TOUCHDET);
137  STMPE610_write_register(STMPE_ADC_CTRL1, STMPE_ADC_CTRL1_10BIT | (0x6 << 4));
138  STMPE610_write_register(STMPE_ADC_CTRL2, STMPE_ADC_CTRL2_6_5MHZ);
139  STMPE610_write_register(STMPE_TSC_CFG, STMPE_TSC_CFG_4SAMPLE | STMPE_TSC_CFG_DELAY_1MS | STMPE_TSC_CFG_SETTLE_5MS);
140  STMPE610_write_register(STMPE_TSC_FRACTION_Z, 0x6);
141  STMPE610_write_register(STMPE_FIFO_TH, 1);
142  STMPE610_write_register(STMPE_FIFO_STA, STMPE_FIFO_STA_RESET);
143  STMPE610_write_register(STMPE_FIFO_STA, 0);
144  STMPE610_write_register(STMPE_TSC_I_DRIVE, STMPE_TSC_I_DRIVE_50MA);
145  STMPE610_write_register(STMPE_INT_STA, 0xFF);
146  STMPE610_write_register(STMPE_INT_CTRL, STMPE_INT_CTRL_POL_HIGH | STMPE_INT_CTRL_ENABLE);
147 
148  return 1;
149 }

168 void STMPE610_get_XYZ (uint8_t *x, uint8_t *y, uint8_t *z)
169 {
170  uint8_t temp_data[4];
171  uint8_t i;
172 
173  for (i = 0; i < 4; i++)
174  {
175  temp_data[i] = STMPE610_read_register(0xD7);
176  }
177 
178  *x = temp_data[0];
179  *x <<= 4;
180  *x |= (temp_data[1] >> 4);
181  *y = temp_data[1] & 0x0F;
182  *y <<= 8;
183  *y |= temp_data[2];
184  *z = temp_data[3];
185 }
