59 void PORT_init (void)
60 {
74  /* Port Configuration for LPSPI0 LCD Control */
75  PCC -> PCCn[PCC_PORTB_INDEX] |= PCC_PCCn_CGC_MASK; /* CGC=1: Clock enabled for PORTB */
76 
77  PORTB -> PCR[PTB2] |= PORT_PCR_MUX(3); /* MUX=3: Select LPSPI0_SCK on PTB2 */
78  PORTB -> PCR[PTB3] |= PORT_PCR_MUX(3); /* MUX=3: Select LPSPI0_SIN on PTB3 */
79  PORTB -> PCR[PTB4] |= PORT_PCR_MUX(3); /* MUX=3: Select LPSPI0_SOUT on PTB4 */
80 
81  PORTB -> PCR[PTB5] |= PORT_PCR_MUX(1); /* MUX=1: Select GPIO on PTB5 */
82  PTB -> PSOR |= 1<<PTB5; /* Assert LCD Chip Select */
83  PTB -> PDDR |= 1<<PTB5; /* Direction: PTB5 as output */
84 
85  /* Port Configuration for Other LCD Control (RS) */
86  PCC -> PCCn[PCC_PORTD_INDEX] |= PCC_PCCn_CGC_MASK; /* CGC=1: Clock enabled for PORTD */
87 
88  PORTD -> PCR[PTD14] |= PORT_PCR_MUX(1); /* MUX=1: Select GPIO on PTD14 */
89  PTD -> PCOR |= 1<<PTD14; /* Clear PTD14 */
90  PTD -> PDDR |= 1<<PTD14; /* PTD14 is an output */
91 
92  /* Port Configuration for Touch CS */
93  PORTD -> PCR[PTD13] |= PORT_PCR_MUX(1); /* MUX=1: Select GPIO on PTD13 */
94  PTD -> PSOR |= 1<<PTD13; /* Assert TOUCH Chip Select */
95  PTD -> PDDR |= 1<<PTD13; /* PTD13 is also an output */
96 }
