39 void WDOG_init (void)
40 {
41  WDOG -> CNT = 0xD928C520; /* Unlock WDOG to reconfigure. See 'Unlocking the Watchdog' in RF */
42 
43  while(((WDOG -> CS & WDOG_CS_ULK_MASK) >> WDOG_CS_ULK_SHIFT) == 0); /* Wait until registers are unlocked */
44 
45  WDOG -> TOVAL = 5000; /* Establish WDOG timeout in (5000*256) * (1/128 kHz) = 10s */
46 
47  WDOG -> WIN = 2500; /* Establish the earliest time after the refresh is valid in 2500 * (1/128 kHz) = 5s */
48 
49 
50  WDOG -> CS |= WDOG_CS_WIN_MASK /* [15] WIN = 1 Enable window mode */
51  | WDOG_CS_CMD32EN_MASK /* [13] CMD32EN = 1 Enable support for 32-bit refresh/unlock command write words */
52  | WDOG_CS_PRES_MASK /* [12] PRES = 1 Enable 256 prescaler */
53  | WDOG_CS_CLK(0b01) /* [9-8] CLK = 0b01 Select clock source as LPO = 128 KHz. See 'Peripheral Module Clocking' in RF */
54  | WDOG_CS_EN_MASK /* [7] EN = 1 Enable WDOG */
55  | WDOG_CS_INT_MASK /* [6] INT = 1 Enable WDOG interrupts */
56  | WDOG_CS_UPDATE_MASK; /* [5] UPDATE = 1 Allow updates */
57  /* DBG, WAIT, STOP = 0 WDOG disable in chip debug/wait/stop modes */
58 
59  while(((WDOG -> CS & WDOG_CS_RCS_MASK) >> WDOG_CS_RCS_SHIFT) == 0); /* Wait until new configuration takes effect */
60 
61  RCM -> SRIE |= RCM_SRIE_WDOG_MASK /* Enable WDOG interrupt */
62  | RCM_SRIE_DELAY(0b11); /* Reset Delay Time: 514 LPO cycles = ~4ms */
63 }
