133 void PORTC_IRQHandler (void)
134 {
135  if((PTC -> PDIR & (1<<PTC12)) == (1<<PTC12)) /* The button is pressed */
136  {
137  PORTC -> PCR[PTC12] |= PORT_PCR_ISF_MASK; /* Clear Interrupt Status Flag */
138 
139  /* If the refresh happens between the allowed time */
140  if(((WDOG -> CNT) < (WDOG -> TOVAL)) && ((WDOG -> CNT) > (WDOG -> WIN)))
141  {
142  PTD -> PSOR |= 1<<PTD15; /* Turn-Off RED_LED */
143  PTD -> PCOR |= 1<<PTD16; /* Turn-On GREEN_LED */
144  LPUART1_transmit_string("\rWDOG refresh was successful! Remember to refresh it periodically!\n\n\r");
145  }
146 
147  /* If the refresh happens before the allowed time */
148  if((WDOG -> CNT) < (WDOG -> WIN)){
149  PTD -> PSOR |= 1<<PTD16; /* Turn-Off GREEN_LED */
150  PTD -> PCOR |= 1<<PTD15; /* Turn-On RED_LED */
151  LPUART1_transmit_string("\rWDOG refresh was too early! Try again!");
152  S32_NVIC -> ICER[WDOG_EWM_IRQn/32] |= (1<<(WDOG_EWM_IRQn%32));
153 
154  }
155 
156  RCM -> SRIE &= ~RCM_SRIE_GIE_MASK; /* Disable global interrupts */
157  WDOG -> CNT = 0xB480A602; /* Refresh WDOG. See "Refreshing the Watchdog" in RF */
158  RCM -> SRIE |= RCM_SRIE_GIE_MASK; /* Enable global interrupts */
159  }
