93  PCC -> PCCn[PCC_PORTC_INDEX] = PCC_PCCn_CGC_MASK; /* Enable clock for PORT C */
94  PCC -> PCCn[PCC_PORTE_INDEX] = PCC_PCCn_CGC_MASK; /* Enable clock for PORT E */

96  PORTC -> PCR[PTC12] = PORT_PCR_MUX(1) /* Port C12: MUX = GPIO */

101  PORTC -> PCR[PTC13] = PORT_PCR_MUX(1) /* Port C13: MUX = GPIO */
102  | PORT_PCR_PFE_MASK; /* Port C13: MUX = GPIO, input filter enabled */
103 
104  PTC -> PDDR &= ~(1<<PTC13); /* Port C13: Data Direction = input */
105 
106  PORTC -> PCR[PTC6] |= PORT_PCR_MUX(2); /* Port C6: MUX = UART1 RX */
107  PORTC -> PCR[PTC7] |= PORT_PCR_MUX(2); /* Port C7: MUX = UART1 TX */
108 
109  PORTE -> PCR[PTE4] |= PORT_PCR_MUX(5); /* Port E4: MUX = CAN0 RX */
